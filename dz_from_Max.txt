void shellsort(int *A, int N){
	int d, n = 10, i, j, t;
	for (d = n / 2; d > 0; d /= 2){
		for (i = d; i < n; i++){
			for (j = i; j >= d && A[j - d] > A[j]; j -= d){
				t = A[j];
				A[j] = A[j - d];
				A[j - d] = t;
			}
		}
	}
}
